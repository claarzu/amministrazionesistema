CREATE TABLE amministratore (    
id INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL PRIMARY KEY,    
nome VARCHAR(128),    
cognome VARCHAR(128),    
username VARCHAR(128),    
password VARCHAR(32)    
);

CREATE TABLE cliente (    
idCliente INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL PRIMARY KEY,    
nomeCliente VARCHAR(128),    
cognomeCliente VARCHAR(128),
indirizzoCliente VARCHAR(128),
capCliente INTEGER,
cittaCliente VARCHAR(128),
emailCliente VARCHAR(128),    
usernameCliente VARCHAR(128),    
passwordCliente VARCHAR(32),    
c_passwordCliente VARCHAR(32) 
);

CREATE TABLE prodotti (
id INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL PRIMARY KEY,
tipo VARCHAR(32),
prezzo DOUBLE NOT NULL
);

CREATE TABLE carrello (
id INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
clienteId INTEGER,
FOREIGN KEY (clienteID) REFERENCES cliente(idCliente),
prodottoId INTEGER,
FOREIGN KEY (prodottoId) REFERENCES prodotti(id),
PRIMARY KEY (id, clienteId, prodottoId) 
);
